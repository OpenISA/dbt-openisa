99  static size_t disassemble(const void* func, std::ostream &buffer) {
100  char outline[1024];
101  size_t Size = 0;
102  uint64_t pc;
103  const uint64_t extent = 96*1024;
104  const uint8_t *bytes = (const uint8_t *) func;
105 
106  //Initialize LLVM targets
107  LLVMInitializeNativeAsmParser();
108  LLVMInitializeNativeAsmPrinter();
109  LLVMInitializeNativeDisassembler();
110  LLVMInitializeNativeTarget();
111 
112  LLVMDisasmContextRef D = LLVMCreateDisasm(LLVM_HOST_TRIPLE, NULL, 0, NULL, NULL);
113 
114  if (!D) {
115  buffer << "Error: Could not create disassembler for triple " << LLVM_HOST_TRIPLE << '\n';
116  return -1;
117  }
118 
119  for(pc = 0; pc<extent; pc+=Size) {
120  unsigned int i;
121  memset(outline, 0, sizeof(outline));
122  // Print address. We use addresses relative to the start of the function, so that between runs.
123  buffer << "<" << std::hex << static_cast<const void*>(bytes) << "+" << std::hex << std::setw(4) << std::setfill('0') << (unsigned long) pc << ">:\t";
124  Size = LLVMDisasmInstruction(D, (uint8_t *)bytes + pc, extent - pc, 0, outline, sizeof (outline));
125 
126  if (!Size) {
127  buffer << "invalid at \n" << pc++;
128  break;
129  }
130  //Output the bytes in hexidecimal format.
131  for (i = 0; i < Size; ++i) {
132  buffer << std::hex << std::setfill('0') << std::setw(2) << static_cast<int> (bytes[pc + i]);
133  }
134 
135  buffer << std::setfill(' ') << std::setw(16-Size) << " ";
136 
137  //Print the instruction.
138  buffer << outline << '\n';
139 
140  if(static_cast<int>(bytes[pc]) == 0xc3) {
141  buffer << "RETURN INSTRUCTION BREAK (0xc3)" << std::endl;
142  break;
143  }
144  }
145 
146  if (pc >= extent) buffer << "disassembly larger than " << extent << " bytes\n";
147 
148  buffer << '\n';
149 
150  LLVMDisasmDispose(D);
151 
152  // Print GDB command, useful to verify output.
153  buffer << "disassemble " << static_cast<const void*>(bytes) << ' ' << static_cast<const void*>(bytes + pc) << '\n';
154 
155  return pc;
156  }
